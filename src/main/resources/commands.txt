docker run -d --name cine-reserve -p 5432:5432 --volume cine-reserve-db-data:/var/lib/postgresql/data -e POSTGRES_USER=postgres -e POSTGRES_PASSWORD=admin -e POSTGRES_DB=cine_reserve_theatres postgres -c max_connections=200
